<script>
    [span_3](start_span)// تم الإبقاء على منطق الاختبار الأصلي والبيانات [cite: 147-173]
    const LEGAL_SIGN = "أ / عبدالعزيز الفيفي - ثانوية الأمير عبدالمجيد";
    function verifyRights() {
        const el = document.getElementById('owner-tag');
        if (!el || el.innerText.trim() !== LEGAL_SIGN) {
            document.getElementById('safety-overlay').style.display = 'flex';
            document.getElementById('quiz-container').style.display = 'none';
        }
    }
    setInterval(verifyRights, 1000);

    const rawData = [
        { q: "ريش : طائر", opts: ["شعر : ثدييات", "جلد : إنسان", "وبر : إبل", "صوف : غنم"], ans: "شعر : ثدييات", tip: "العلاقة (غطاء الجسم للكائن)؛ الريش يغطي الطائر، والشعر يغطي الثدييات كفئة عامة.", cat: "تناظر" },
        // ... (بقية الأسئلة 149-173 تبقى كما هي في ملفك الأصلي)
        { q: "الارتباط: 'فصل : مدرسة' مثل 'غرفة : ........'", opts: ["فندق", "شارع", "حديقة", "مطعم"], ans: "فندق", tip: "علاقة جزء من كل؛ الغرفة جزء من الفندق.", cat: "ارتباط" }
    ];

    // جعل المتغيرات عالمية لتصل إليها دالة Firebase
    window.currentQ = 0; window.scoreCount = 0; window.logsList = []; window.sName = "";
    let timerObj, mixedData = [];
    
    function shuffle(arr) { return arr.sort(() => Math.random() - 0.5); }

    function startQuiz() {
        window.sName = document.getElementById('userNameInput').value;
        if(!window.sName.trim()) return alert("لطفاً، أدخل اسمك.");
        document.getElementById('display-name').innerText = window.sName;
        document.getElementById('start-screen').classList.add('hidden');
        document.getElementById('quiz-screen').classList.remove('hidden');
        mixedData = shuffle([...rawData]);
        loadQuestion();
    }

    function loadQuestion() {
        verifyRights();
        if(window.currentQ >= mixedData.length) return endQuiz();
        const item = mixedData[window.currentQ];
        document.getElementById('display-q-idx').innerText = `السؤال ${window.currentQ + 1} / 40`;
        document.getElementById('display-question').innerText = item.q;
        document.getElementById('feedback-panel').style.display = 'none';
        document.getElementById('next-btn').classList.add('hidden');
        
        const txtBox = document.getElementById('display-text');
        if(item.text) { txtBox.innerText = item.text; txtBox.classList.remove('hidden'); }
        else { txtBox.classList.add('hidden'); }

        const optBox = document.getElementById('display-options');
        optBox.innerHTML = '';
        shuffle([...item.opts]).forEach(o => {
            const b = document.createElement('div');
            b.className = 'opt-btn';
            b.innerText = o;
            b.onclick = () => checkResult(o, b);
            optBox.appendChild(b);
        });
        runTimer();
    }

    function checkResult(sel, el) {
        clearInterval(timerObj);
        const item = mixedData[window.currentQ];
        const isOk = (sel === item.ans);
        document.querySelectorAll('.opt-btn').forEach(b => {
            b.onclick = null;
            if(b.innerText === item.ans) b.classList.add('correct');
            else if(b.innerText === sel) b.classList.add('wrong');
        });
        if(isOk) window.scoreCount++;
        const p = document.getElementById('feedback-panel');
        p.style.display = 'block';
        document.getElementById('fb-title').innerText = isOk ? "رائع! إجابة دقيقة ✅" : "للأسف، إجابة خاطئة ❌";
        document.getElementById('fb-desc').innerText = item.tip;
        document.getElementById('next-btn').classList.remove('hidden');
        
        [cite_start]// حفظ تفاصيل الإجابة للتحليل اللاحق[span_3](end_span)
        window.logsList.push({ 
            q: item.q, 
            cat: item.cat, 
            state: isOk ? 'ok' : 'no', 
            userAns: sel, 
            correctAns: item.ans, 
            tip: item.tip 
        });
    }

    function runTimer() {
        let t = 45;
        const d = document.getElementById('display-timer');
        d.innerText = t;
        d.classList.remove('timer-critical');
        timerObj = setInterval(() => {
            t--;
            d.innerText = t;
            if(t <= 10) d.classList.add('timer-critical');
            if(t <= 0) { clearInterval(timerObj); checkResult("تخطى الوقت", null); }
        }, 1000);
    }

    function handleNext() { window.currentQ++; loadQuestion(); }

    function endQuiz() {
        document.getElementById('quiz-screen').classList.add('hidden');
        document.getElementById('result-screen').classList.remove('hidden');
        document.getElementById('res-score').innerText = `${window.scoreCount} / 40`;
        document.getElementById('res-msg').innerText = `تهانينا يا ${window.sName}! لقد أتممت جميع النماذج بنجاح.`;
        
        const box = document.getElementById('res-logs');
        box.innerHTML = '';
        window.logsList.forEach((l, i) => {
            const div = document.createElement('div');
            div.className = `log-item ${l.state==='ok'?'log-ok':'log-no'}`;
            div.innerHTML = `<div class="watermark">© أ/ عبدالعزيز الفيفي</div><strong>س${i+1}: [${l.cat}] ${l.q}</strong><br>إجابتك: ${l.userAns}<br>الجواب النموذجي: ${l.correctAns}<br><div style="margin-top:10px; font-size:0.9rem; color:#444; border-top:1px solid #eee; padding-top:8px;">💡 الشرح التعليمي: ${l.tip}</div>`;
            box.appendChild(div);
        });
    }
